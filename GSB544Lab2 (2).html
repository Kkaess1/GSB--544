import pandas as pd
from plotnine import *

grouped = (
    df_ca.groupby(["geography", "type"])[["S_HASS", "L_HASS", "XL_HASS"]]
    .mean()
    .reset_index())

df_long = grouped.melt(
    id_vars=["geography", "type"],
    value_vars=["S_HASS", "L_HASS", "XL_HASS"],
    var_name="Size",
    value_name="Sales")

df_long["Proportion"] = df_long["Sales"] / df_long.groupby(["geography", "type"])["Sales"].transform("sum")

final_graph = (
    ggplot(df_long, aes(x="geography", y="Proportion", fill="Size"))
    + geom_bar(stat="identity", position="stack")
    + facet_wrap("~type")
    + labs(
        title="Proportion of Average Hass Avocado Sales by Size",
        x="Region of California",
        y="Proportion of Sales")
)
